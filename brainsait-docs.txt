# scripts/generation/doc_generator.py
"""
BrainSAIT Auto-Documentation System
Powered by MasterLinc orchestration
"""

from typing import Dict, List, Optional
from dataclasses import dataclass
from pathlib import Path
import yaml
import anthropic

@dataclass
class DocumentSpec:
    """Document generation specification"""
    domain: str  # healthcare, business, tech, personal
    chapter: str
    section: str
    source_type: str  # code, transcript, notes, manual
    source_path: Optional[Path] = None
    template: str = "standard"
    
class BrainSAITDocGenerator:
    """
    Automated documentation generation system
    
    BRAINSAIT: Audit logging enabled
    AGENT: MasterLinc orchestration
    BILINGUAL: AR/EN output
    """
    
    def __init__(self, config_path: str):
        self.config = self._load_config(config_path)
        self.client = anthropic.Client()
        
    def generate_healthcare_docs(self, specs: List[DocumentSpec]) -> Dict[str, str]:
        """
        Generate healthcare documentation from specifications
        
        Args:
            specs: List of document specifications
            
        Returns:
            Dict mapping output paths to generated content
            
        MEDICAL: FHIR/clinical validation
        """
        results = {}
        
        for spec in specs:
            # Route to appropriate agent
            if spec.section == "claims":
                agent = "ClaimLinc"
            elif spec.section == "nphies":
                agent = "PolicyLinc"
            elif spec.section == "sop":
                agent = "ProcessLinc"
            else:
                agent = "MasterLinc"
            
            content = self._generate_with_agent(spec, agent)
            output_path = self._build_output_path(spec)
            results[output_path] = content
            
        return results
    
    def _generate_with_agent(self, spec: DocumentSpec, agent: str) -> str:
        """Generate content using specified agent"""
        
        prompt = f"""Generate {spec.domain} documentation for:

Chapter: {spec.chapter}
Section: {spec.section}
Template: {spec.template}

Requirements:
- Follow BrainSAIT documentation standards
- Include bilingual headers (EN/AR)
- Use technical accuracy for healthcare content
- Include code examples where relevant
- Add FHIR validation rules for clinical content
- Include audit logging examples
- Reference BrainSAIT agents appropriately

Format: Markdown with YAML frontmatter"""

        if spec.source_path:
            with open(spec.source_path) as f:
                source_content = f.read()
            prompt += f"\n\nSource Material:\n{source_content}"
        
        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=8000,
            messages=[{"role": "user", "content": prompt}]
        )
        
        return response.content[0].text
    
    def _build_output_path(self, spec: DocumentSpec) -> str:
        """Build output file path from spec"""
        return f"docs/{spec.domain}/{spec.chapter}/{spec.section}.md"

Index documentation for Efhm RAG system

from pathlib import Path
import chromadb
from typing import List, Dict

class BrainSAITDocsIndexer:
    """
    Index documentation for RAG retrieval
    
    AGENT: Efhm integration
    NEURAL: Embedding generation
    """
    
    def __init__(self, docs_path: Path, collection_name: str = "brainsait_docs"):
        self.docs_path = docs_path
        self.client = chromadb.Client()
        self.collection = self.client.get_or_create_collection(
            name=collection_name,
            metadata={"description": "BrainSAIT Master Documentation"}
        )
    
    def index_all_docs(self):
        """Index all markdown documents"""
        
        documents = []
        metadatas = []
        ids = []
        
        for md_file in self.docs_path.rglob("*.md"):
            content, metadata = self._parse_markdown(md_file)
            
            documents.append(content)
            metadatas.append(metadata)
            ids.append(str(md_file.relative_to(self.docs_path)))
        
        self.collection.add(
            documents=documents,
            metadatas=metadatas,
            ids=ids
        )
        
        print(f"Indexed {len(documents)} documents")
    
    def _parse_markdown(self, file_path: Path) -> tuple[str, Dict]:
        """Parse markdown with YAML frontmatter"""
        
        with open(file_path) as f:
            content = f.read()
        
        # Split frontmatter and content
        if content.startswith("---"):
            parts = content.split("---", 2)
            frontmatter = yaml.safe_load(parts[1])
            body = parts[2].strip()
        else:
            frontmatter = {}
            body = content
        
        metadata = {
            "domain": frontmatter.get("domain", "unknown"),
            "chapter": frontmatter.get("chapter", ""),
            "section": frontmatter.get("section", ""),
            "agents": frontmatter.get("agents", []),
            "compliance": frontmatter.get("compliance", []),
            "path": str(file_path)
        }
        
        return body, metadata

python
# AGENT: PolicyLinc architecture
# BILINGUAL: Arabic policy interpretation
# NEURAL: RAG-based policy retrieval

from typing import List, Optional
from dataclasses import dataclass
import anthropic
from brainsait.rag import PolicyKnowledgeBase

@dataclass
class PolicyRule:
    """Structured policy rule"""
    rule_id: str
    payer_id: str
    service_category: str
    coverage_conditions: List[str]
    exclusions: List[str]
    prior_auth_required: bool
    copayment_percentage: Optional[float]
    max_visits: Optional[int]
    effective_date: str
    expiry_date: Optional[str]
    source_document: str
    confidence_score: float

class PolicyLinc:
    """
    Payer policy interpretation and coverage guidance
    
    AGENT: LLM-powered policy analysis
    BILINGUAL: Arabic/English support
    """
    
    def __init__(self, config: PolicyLincConfig):
        self.llm_client = anthropic.Client()
        self.knowledge_base = PolicyKnowledgeBase()
        self.payer_configs = self._load_payer_configs()
    
    async def ingest_policy_document(
        self,
        payer_id: str,
        document_path: str,
        document_type: str = "policy_manual"
    ) -> IngestionResult:
        """
        Ingest and structure policy document
        
        Args:
            payer_id: Insurance company identifier
            document_path: Path to policy document
            document_type: Type of document
            
        Returns:
            Ingestion results with extracted rules
            
        BILINGUAL: Handles Arabic policy documents
        """
        
        # Extract text from document
        document_text = await self._extract_text(document_path)
        
        # Detect language
        language = self._detect_language(document_text)
        
        # Extract policy rules using LLM
        extraction_prompt = f"""Extract structured policy rules from this {language} insurance policy document.

Document Type: {document_type}
Payer: {payer_id}

Document Content:
{document_text}

For each coverage rule, extract:
1. Service category (e.g., "Diagnostic Imaging", "Surgical Procedures")
2. Coverage conditions (when service is covered)
3. Exclusions (when service is NOT covered)
4. Prior authorization requirement (yes/no)
5. Copayment percentage (if specified)
6. Visit limits (if specified)
7. Age restrictions (if any)
8. Network restrictions (if any)

Output as JSON array of rules with both English and Arabic descriptions."""

        llm_response = self.llm_client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=8000,
            messages=[{"role": "user", "content": extraction_prompt}]
        )
        
        extracted_rules = json.loads(llm_response.content[0].text)
        
        # Structure and validate rules
        policy_rules = []
        for rule_data in extracted_rules:
            rule = PolicyRule(
                rule_id=self._generate_rule_id(payer_id, rule_data),
                payer_id=payer_id,
                service_category=rule_data["service_category"],
                coverage_conditions=rule_data["coverage_conditions"],
                exclusions=rule_data["exclusions"],
                prior_auth_required=rule_data["prior_auth_required"],
                copayment_percentage=rule_data.get("copayment_percentage"),
                max_visits=rule_data.get("max_visits"),
                effective_date=datetime.now().isoformat(),
                expiry_date=None,
                source_document=document_path,
                confidence_score=rule_data.get("confidence", 0.9)
            )
            policy_rules.append(rule)
        
        # Index in knowledge base
        await self.knowledge_base.index_rules(policy_rules)
        
        # Audit log
        audit_logger.log_policy_ingestion(
            payer_id=payer_id,
            document_path=document_path,
            rules_extracted=len(policy_rules),
            language=language
        )
        
        return IngestionResult(
            payer_id=payer_id,
            rules_extracted=len(policy_rules),
            language=language,
            policy_rules=policy_rules
        )
    
    async def check_coverage(
        self,
        member_id: str,
        payer_id: str,
        service_code: str,
        service_date: datetime,
        diagnosis_codes: List[str] = None
    ) -> CoverageResult:
        """
        Check coverage for specific service
        
        AGENT: RAG retrieval + LLM reasoning
        MEDICAL: Clinical context consideration
        """
        
        # Retrieve relevant policy rules
        relevant_rules = await self.knowledge_base.retrieve_rules(
            payer_id=payer_id,
            service_code=service_code,
            diagnosis_codes=diagnosis_codes
        )
        
        # Check eligibility via NPHIES
        eligibility = await self._check_nphies_eligibility(
            member_id=member_id,
            payer_id=payer_id,
            service_date=service_date
        )
        
        if not eligibility.is_active:
            return CoverageResult(
                is_covered=False,
                reason="Member not eligible",
                eligibility=eligibility
            )
        
        # Analyze coverage using LLM
        coverage_prompt = f"""Determine if this service is covered under the policy:

Member ID: {member_id}
Payer: {payer_id}
Service Code: {service_code}
Service Date: {service_date}
Diagnosis Codes: {diagnosis_codes}

Relevant Policy Rules:
{json.dumps([r.__dict__ for r in relevant_rules], indent=2)}

Eligibility Status:
{json.dumps(eligibility.__dict__, indent=2)}

Determine:
1. Is service covered? (yes/no/conditional)
2. Coverage percentage
3. Prior authorization required?
4. Exclusions that apply?
5. Special conditions?
6. Patient responsibility estimate

Provide detailed reasoning in English and Arabic."""

        llm_response = self.llm_client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=3000,
            messages=[{"role": "user", "content": coverage_prompt}]
        )
        
        analysis = json.loads(llm_response.content[0].text)
        
        return CoverageResult(
            is_covered=analysis["is_covered"],
            coverage_percentage=analysis.get("coverage_percentage", 0),
            prior_auth_required=analysis["prior_auth_required"],
            exclusions=analysis.get("exclusions", []),
            conditions=analysis.get("conditions", []),
            patient_responsibility=analysis.get("patient_responsibility"),
            reasoning=analysis["reasoning"],
            applicable_rules=relevant_rules,
            eligibility=eligibility
        )
    
    async def detect_policy_changes(
        self,
        payer_id: str,
        new_document_path: str,
        old_document_path: str
    ) -> PolicyChangeReport:
        """
        Compare policy versions and detect changes
        
        AGENT: Change detection
        """
        
        # Ingest both documents
        old_rules = await self.ingest_policy_document(
            payer_id=payer_id,
            document_path=old_document_path
        )
        
        new_rules = await self.ingest_policy_document(
            payer_id=payer_id,
            document_path=new_document_path
        )
        
        # Compare rules
        changes = self._compare_policy_rules(
            old_rules.policy_rules,
            new_rules.policy_rules
        )
        
        # Assess impact
        impact_analysis = await self._analyze_change_impact(changes)
        
        return PolicyChangeReport(
            payer_id=payer_id,
            changes=changes,
            impact_analysis=impact_analysis,
            affected_claims_estimate=impact_analysis.affected_claims
        )
